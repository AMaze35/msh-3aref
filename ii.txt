# Simple Login System with Dictionary

database = {}
User = "Username: "

def CheckUsername(x):
    return x in database.keys()

def Login():
    x = input(User)
    if not CheckUsername(x):
        return 0
    y = input("Password: ")
    if y == database[x]["password"]:
        return x
    else:
        return -1
    
def Register():
    x = input(User)
    if CheckUsername(x):
        return -1
    password = input("Password: ")
    secret = input("Now enter your secret phrase for safekeeping: ")
    database[x] = {"password": password, "secret": secret}
    return 1

def Choices():
    print("\nWhat would you like to do?")
    print("1) Login")
    print("2) Register")
    print("3) Exit\n")
    ans = int(input("> "))
    print()
    return ans

while True:
    result = Choices()

    if result == 1:
        log = Login()

        if log == 0:
            print("Error: Username Not Found")

        elif log == -1:
            print("Error: Incorrect Password")

        else:
            print(f"\nWelcome back, {log}!")
            print("Your secret phrase is:")
            print(database[log]["secret"])

        continue

    if result == 2:
        acc = Register()

        if acc == 1:
            print("\nSuccessfuly Registered!")

        elif  acc == -1:
            print("Error: Username Taken")

        continue

    if result == 3:
        print("Thank You")
        break

    else:
        print("error: invalid input")
